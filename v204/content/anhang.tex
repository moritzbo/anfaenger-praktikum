\begin{align}
      \bar{x} &= \frac{1}{\symup{N}} \sum_{i=0}^\text{N} x_i
      \label{eqn:mittelwert} \\
      \increment\overline{x} &= \sqrt{
      \frac{1}{\symup{N}(\symup{N}-1)}\sum_{k=1}^\text{N}
      \left( x_k - \overline{x} \right)^2}
      \label{eqn:mittelwertfehler}
\end{align}
